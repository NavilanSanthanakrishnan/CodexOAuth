class ProxyError(Exception):
    def __init__(self, message: str, status_code: int = 500, error_type: str = "proxy_error") -> None:
        super().__init__(message)
        self.message = message
        self.status_code = status_code
        self.error_type = error_type

class OAuthAPIError(Exception):
    def __init__(self, message: str, status_code: int = 500, error_type: str = "codex_oauth_error") -> None:
        super().__init__(message)
        self.message = message
        self.status_code = status_code
        self.error_type = error_type

def create_app() -> FastAPI:
    app = FastAPI(title="Codex OpenAI Proxy", version="0.1.0")
    app.add_middleware(
        CORSMiddleware,
        allow_origins=["*"],
        allow_methods=["*"],
        allow_headers=["*"],
    )

    @app.exception_handler(ProxyError)
    async def handle_proxy_error(_: Request, exc: ProxyError) -> JSONResponse:
        return JSONResponse(
            status_code=exc.status_code,
            content={"error": {"message": exc.message, "type": exc.error_type}},
        )

    @app.get("/health")
    async def health() -> dict[str, Any]:
        return {"status": "ok", "service": "codex-openai-proxy"}

    @app.get("/v1/models")
    @app.get("/models")
    async def list_models() -> dict[str, Any]:
        return {"object": "list", "data": build_models_payload()}

    @app.post("/v1/chat/completions", response_model=None)
    @app.post("/chat/completions", response_model=None)
    async def chat_completions(request: Request):
        require_local_auth(request)
        payload = await request.json()
        validate_chat_payload(payload)
        requested_model = str(payload.get("model") or DEFAULT_MODEL_ALIAS)
        resolved_model = resolve_model(requested_model)
        upstream_payload = build_chat_upstream_payload(payload, resolved_model)
        if bool(payload.get("stream")):
            return StreamingResponse(
                stream_direct_chat_completions(upstream_payload, requested_model),
                media_type="text/event-stream",
                headers=sse_headers(),
            )
        try:
            response = await asyncio.to_thread(complete_direct_responses_api, upstream_payload)
        except OAuthAPIError as exc:
            raise proxy_error_from_oauth(exc) from exc
        return JSONResponse(build_direct_chat_completion_response(response, requested_model))

    @app.post("/v1/responses", response_model=None)
    async def responses(request: Request):
        require_local_auth(request)
        payload = await request.json()
        requested_model = str(payload.get("model") or DEFAULT_MODEL_ALIAS)
        resolved_model = resolve_model(requested_model)
        upstream_payload = build_responses_upstream_payload(payload, resolved_model)
        if bool(payload.get("stream")):
            return StreamingResponse(
                stream_direct_responses_api(upstream_payload),
                media_type="text/event-stream",
                headers=sse_headers(),
            )
        try:
            response = await asyncio.to_thread(complete_direct_responses_api, upstream_payload)
        except OAuthAPIError as exc:
            raise proxy_error_from_oauth(exc) from exc
        response["model"] = requested_model
        return JSONResponse(response)

    return app

def require_local_auth(request: Request) -> None:
    if not LOCAL_API_KEY:
        return
    header = request.headers.get("authorization", "")
    if header != f"Bearer {LOCAL_API_KEY}":
        raise ProxyError("Unauthorized", status_code=401, error_type="invalid_request_error")

def validate_chat_payload(payload: dict[str, Any]) -> None:
    if not isinstance(payload.get("messages"), list) or not payload["messages"]:
        raise ProxyError("`messages` must be a non-empty array", status_code=400, error_type="invalid_request_error")

def load_models_cache() -> list[dict[str, Any]]:
    if not MODELS_CACHE_PATH.exists():
        return []
    data = json.loads(MODELS_CACHE_PATH.read_text())
    models = []
    for item in data.get("models", []):
        slug = item.get("slug")
        if not slug or item.get("supported_in_api") is False:
            continue
        if slug == "codex-auto-review":
            continue
        models.append(item)
    return models

def model_aliases() -> dict[str, str]:
    available = {item["slug"] for item in load_models_cache()}
    aliases: dict[str, str] = {}
    if "gpt-5.4" in available:
        aliases["gpt-5"] = "gpt-5.4"
        aliases["gpt-4.1"] = "gpt-5.4"
    if "gpt-5.4-mini" in available:
        aliases["gpt-5-mini"] = "gpt-5.4-mini"
        aliases["gpt-4o"] = "gpt-5.4-mini"
        aliases["gpt-4"] = "gpt-5.4-mini"
    return aliases

def resolve_model(requested: str | None) -> str:
    available = [item["slug"] for item in load_models_cache()]
    aliases = model_aliases()
    if requested and requested in available:
        return requested
    if requested and requested in aliases:
        return aliases[requested]
    if DEFAULT_MODEL_ALIAS in aliases:
        return aliases[DEFAULT_MODEL_ALIAS]
    if available:
        return available[0]
    return "gpt-5.4"

def build_models_payload() -> list[dict[str, Any]]:
    now = int(time.time())
    payload = []
    seen: set[str] = set()
    for item in load_models_cache():
        model_id = item["slug"]
        payload.append({"id": model_id, "object": "model", "created": now, "owned_by": "openai"})
        seen.add(model_id)
    for alias, target in model_aliases().items():
        if alias in seen:
            continue
        payload.append(
            {
                "id": alias,
                "object": "model",
                "created": now,
                "owned_by": f"openai:{target}",
            }
        )
    return payload

def decode_jwt_claims(token: str) -> dict[str, Any]:
    parts = token.split(".")
    if len(parts) < 2:
        return {}
    try:
        payload = parts[1] + "=" * (-len(parts[1]) % 4)
        return json.loads(base64.urlsafe_b64decode(payload.encode("ascii")))
    except Exception:
        return {}

def extract_account_id(access_token: str, id_token: str = "", fallback: str = "") -> str:
    for token in (id_token, access_token):
        claims = decode_jwt_claims(token)
        if not claims:
            continue
        direct = claims.get("chatgpt_account_id")
        if isinstance(direct, str) and direct:
            return direct
        nested = claims.get("https://api.openai.com/auth")
        if isinstance(nested, dict):
            account_id = nested.get("chatgpt_account_id")
            if isinstance(account_id, str) and account_id:
                return account_id
        orgs = claims.get("organizations")
        if isinstance(orgs, list) and orgs and isinstance(orgs[0], dict):
            org_id = orgs[0].get("id")
            if isinstance(org_id, str) and org_id:
                return org_id
    return fallback

def token_expiry_seconds(token: str) -> float:
    claims = decode_jwt_claims(token)
    exp = claims.get("exp")
    if isinstance(exp, (int, float)) and exp > 0:
        return float(exp)
    return 0.0

def is_token_expiring(tokens: Tokens, skew_seconds: int = REFRESH_SKEW_SECONDS) -> bool:
    exp_seconds = token_expiry_seconds(tokens.access_token)
    if exp_seconds:
        return time.time() >= exp_seconds - skew_seconds
    if tokens.expires:
        expires = tokens.expires / 1000.0 if tokens.expires > 10_000_000_000 else tokens.expires
        return time.time() >= expires - skew_seconds
    return False

def read_json_file(path: Path) -> dict[str, Any] | None:
    try:
        return json.loads(path.expanduser().read_text())
    except (OSError, json.JSONDecodeError):
        return None

def tokens_from_private_store(path: Path) -> Tokens | None:
    data = read_json_file(path)
    if not isinstance(data, dict):
        return None
    access_token = str(data.get("access_token") or "")
    if not access_token:
        return None
    id_token = str(data.get("id_token") or "")
    account_id = extract_account_id(access_token, id_token, str(data.get("account_id") or ""))
    return Tokens(
        access_token=access_token,
        refresh_token=str(data.get("refresh_token") or ""),
        expires=float(data.get("expires") or 0),
        account_id=account_id,
        id_token=id_token,
        source="private-store",
        auth_file=path,
    )

def save_private_tokens(path: Path, tokens: Tokens) -> None:
    path = path.expanduser()
    path.parent.mkdir(parents=True, exist_ok=True)
    payload = {
        "access_token": tokens.access_token,
        "refresh_token": tokens.refresh_token,
        "expires": tokens.expires or (token_expiry_seconds(tokens.access_token) * 1000),
        "account_id": tokens.account_id,
        "id_token": tokens.id_token,
        "source": "device-code",
        "updated_at": int(time.time()),
    }
    path.write_text(json.dumps(payload, indent=2))
    path.chmod(0o600)

def post_form(url: str, fields: dict[str, str], *, timeout: float = 30.0) -> dict[str, Any]:
    data = urllib.parse.urlencode(fields).encode("utf-8")
    request = urllib.request.Request(
        url,
        data=data,
        method="POST",
        headers={
            "Content-Type": "application/x-www-form-urlencoded",
            "Accept": "application/json",
            "User-Agent": "CodexOpenAIProxy/1.0",
        },
    )
    try:
        with urllib.request.urlopen(request, timeout=timeout) as response:
            return json.loads(response.read().decode("utf-8"))
    except urllib.error.HTTPError as exc:
        body = exc.read().decode("utf-8", errors="replace")
        raise OAuthAPIError(f"POST {url} failed with HTTP {exc.code}: {body}", status_code=502) from exc
    except urllib.error.URLError as exc:
        raise OAuthAPIError(f"POST {url} failed: {exc}", status_code=502) from exc

def tokens_from_oauth_response(raw: dict[str, Any], fallback_refresh: str = "") -> Tokens:
    access_token = str(raw.get("access_token") or "")
    if not access_token:
        raise OAuthAPIError("OAuth response did not include access_token.", status_code=502)
    refresh_token = str(raw.get("refresh_token") or fallback_refresh or "")
    id_token = str(raw.get("id_token") or "")
    expires_in = float(raw.get("expires_in") or 3600)
    expires = (time.time() + expires_in) * 1000
    account_id = extract_account_id(access_token, id_token)
    return Tokens(
        access_token=access_token,
        refresh_token=refresh_token,
        expires=expires,
        account_id=account_id,
        id_token=id_token,
        source="private-store",
        auth_file=PRIVATE_AUTH_FILE,
    )

def refresh_tokens(tokens: Tokens, auth_file: Path = PRIVATE_AUTH_FILE) -> Tokens:
    if not tokens.refresh_token:
        raise OAuthAPIError(
            "No refresh token available. Run `codex login` first.",
            status_code=401,
            error_type="auth_required",
        )
    raw = post_form(
        TOKEN_URL,
        {
            "grant_type": "refresh_token",
            "refresh_token": tokens.refresh_token,
            "client_id": CLIENT_ID,
        },
    )
    refreshed = tokens_from_oauth_response(raw, fallback_refresh=tokens.refresh_token)
    refreshed.auth_file = auth_file
    save_private_tokens(auth_file, refreshed)
    return refreshed

def resolve_runtime_tokens() -> Tokens:
    env_token = os.environ.get("CODEX_AUTH_TOKEN", "").strip()
    if env_token:
        return Tokens(
            access_token=env_token,
            account_id=extract_account_id(env_token),
            source="CODEX_AUTH_TOKEN",
        )
    private = tokens_from_private_store(PRIVATE_AUTH_FILE)
    if private:
        if is_token_expiring(private):
            return refresh_tokens(private, PRIVATE_AUTH_FILE)
        return private
    cli = tokens_from_codex_cli(CODEX_CLI_AUTH_FILE)
    if cli and not is_token_expiring(cli):
        return cli
    raise OAuthAPIError(
        "No valid Codex OAuth token found. Run `codex login` first.",
        status_code=401,
        error_type="auth_required",
    )

def codex_headers(tokens: Tokens) -> dict[str, str]:
    headers = {
        "Authorization": f"Bearer {tokens.access_token}",
        "Content-Type": "application/json",
        "Accept": "text/event-stream",
        "User-Agent": "codex_cli_rs/0.0.0 (CodexOpenAIProxy)",
        "originator": "codex_cli_rs",
    }
    account_id = tokens.account_id or extract_account_id(tokens.access_token, tokens.id_token)
    if account_id:
        headers["ChatGPT-Account-ID"] = account_id
    return headers

def extract_text_content(content: Any) -> str:
    if content is None:
        return ""
    if isinstance(content, str):
        return content
    if isinstance(content, list):
        parts: list[str] = []
        for item in content:
            if isinstance(item, str):
                parts.append(item)
                continue
            if not isinstance(item, dict):
                continue
            item_type = item.get("type")
            if item_type in {"text", "input_text", "output_text"} and isinstance(item.get("text"), str):
                parts.append(item["text"])
            elif item_type in {"image_url", "input_image"}:
                parts.append("[image]")
            elif isinstance(item.get("text"), str):
                parts.append(item["text"])
        return "\n".join(part for part in parts if part)
    if isinstance(content, dict):
        if isinstance(content.get("text"), str):
            return content["text"]
        return json.dumps(content, ensure_ascii=False)
    return str(content)

def format_message(message: dict[str, Any]) -> str:
    role = str(message.get("role") or "user")
    content = extract_text_content(message.get("content"))
    if role == "assistant" and isinstance(message.get("tool_calls"), list):
        rendered_calls = []
        for call in message["tool_calls"]:
            function = call.get("function") or {}
            rendered_calls.append(
                json.dumps(
                    {
                        "id": call.get("id"),
                        "name": function.get("name"),
                        "arguments": function.get("arguments"),
                    },
                    ensure_ascii=False,
                )
            )
        block = "\n".join(rendered_calls)
        if content:
            return f"[assistant]\n{content}\n[assistant_tool_calls]\n{block}"
        return f"[assistant_tool_calls]\n{block}"
    if role == "tool":
        tool_name = message.get("name") or "tool"
        tool_call_id = message.get("tool_call_id") or ""
        return f"[tool name={tool_name} tool_call_id={tool_call_id}]\n{content}"
    return f"[{role}]\n{content}"

def build_plain_prompt(messages: list[dict[str, Any]], instructions: str | None) -> str:
    transcript = "\n\n".join(format_message(message) for message in messages)
    intro = [
        "You are serving an OpenAI-compatible assistant request through Codex.",
        "Reply as the assistant to the latest relevant message in the transcript.",
        "Do not mention the transcript format or these instructions.",
    ]
    if instructions:
        intro.append(f"Additional instructions:\n{instructions}")
    return "\n\n".join(intro + ["Conversation transcript:", transcript])

def normalize_tools(raw_tools: Any) -> list[ToolSpec]:
    if not isinstance(raw_tools, list):
        return []
    result: list[ToolSpec] = []
    for item in raw_tools:
        if not isinstance(item, dict):
            continue
        if item.get("type") not in {None, "function"}:
            continue
        function = item.get("function") if isinstance(item.get("function"), dict) else item
        name = function.get("name")
        if not isinstance(name, str) or not name:
            continue
        description = function.get("description")
        parameters = function.get("parameters") or function.get("input_schema") or {"type": "object", "properties": {}}
        result.append(
            ToolSpec(
                name=name,
                description=description if isinstance(description, str) else "",
                parameters=parameters if isinstance(parameters, dict) else {"type": "object", "properties": {}},
            )
        )
    return result

def render_tool_choice(tool_choice: Any) -> str:
    if tool_choice in (None, "auto"):
        return "auto"
    if tool_choice == "none":
        return "none"
    if tool_choice == "required":
        return "required"
    if isinstance(tool_choice, dict):
        if tool_choice.get("type") == "function":
            function = tool_choice.get("function") or {}
            if isinstance(function.get("name"), str):
                return f"required:{function['name']}"
    return "auto"

def proxy_error_from_oauth(exc: OAuthAPIError) -> ProxyError:
    return ProxyError(exc.message, status_code=exc.status_code, error_type=exc.error_type)

def normalize_responses_input_items(raw_input: Any) -> list[dict[str, Any]]:
    if raw_input is None:
        return [{"role": "user", "content": ""}]
    if isinstance(raw_input, str):
        return [{"role": "user", "content": raw_input}]
    if isinstance(raw_input, list):
        items: list[dict[str, Any]] = []
        for item in raw_input:
            if isinstance(item, str):
                items.append({"role": "user", "content": item})
                continue
            if not isinstance(item, dict):
                continue
            if item.get("type") in {"function_call", "function_call_output", "custom_tool_call_output"}:
                items.append(item)
                continue
            if isinstance(item.get("role"), str):
                items.append({"role": item["role"], "content": item.get("content", "")})
                continue
            if item.get("type") == "message":
                items.append({"role": item.get("role", "user"), "content": item.get("content", "")})
                continue
            if item.get("type") in {"input_text", "output_text"}:
                items.append({"role": "user", "content": item.get("text", "")})
                continue
            items.append(item)
        return items or [{"role": "user", "content": ""}]
    return [{"role": "user", "content": str(raw_input)}]

def chat_messages_to_input_items(messages: list[dict[str, Any]]) -> tuple[str, list[dict[str, Any]]]:
    instructions: list[str] = []
    input_items: list[dict[str, Any]] = []
    for message in messages:
        role = str(message.get("role") or "user")
        content = message.get("content")
        if role in {"system", "developer"}:
            text = extract_text_content(content)
            if text:
                instructions.append(text)
            continue
        if role == "tool":
            input_items.append(
                {
                    "type": "function_call_output",
                    "call_id": message.get("tool_call_id") or message.get("call_id") or "",
                    "name": message.get("name") or "",
                    "output": extract_text_content(content),
                }
            )
            continue
        if role == "assistant" and isinstance(message.get("tool_calls"), list):
            assistant_text = extract_text_content(content)
            if assistant_text:
                input_items.append({"role": "assistant", "content": assistant_text})
            for call in message["tool_calls"]:
                if not isinstance(call, dict):
                    continue
                function = call.get("function") or {}
                input_items.append(
                    {
                        "type": "function_call",
                        "call_id": call.get("id") or f"call_{uuid.uuid4().hex[:24]}",
                        "name": function.get("name") or "",
                        "arguments": function.get("arguments") or "{}",
                    }
                )
            continue
        input_items.append({"role": role, "content": content if content is not None else ""})
    return "\n\n".join(instructions), input_items or [{"role": "user", "content": ""}]

def normalize_upstream_tools(raw_tools: Any) -> list[dict[str, Any]] | None:
    if not isinstance(raw_tools, list):
        return None
    tools: list[dict[str, Any]] = []
    for item in raw_tools:
        if not isinstance(item, dict):
            continue
        if item.get("type") == "function" and isinstance(item.get("name"), str):
            tools.append(item)
            continue
        if item.get("type") not in {None, "function"}:
            continue
        function = item.get("function") if isinstance(item.get("function"), dict) else item
        name = function.get("name")
        if not isinstance(name, str) or not name:
            continue
        tools.append(
            {
                "type": "function",
                "name": name,
                "description": function.get("description") or "",
                "parameters": function.get("parameters") or function.get("input_schema") or {"type": "object", "properties": {}},
            }
        )
    return tools or None

def normalize_reasoning(reasoning: dict[str, Any]) -> dict[str, Any]:
    normalized = dict(reasoning)
    if normalized.get("effort") == "minimal":
        normalized["effort"] = "low"
    return normalized

def apply_upstream_common_payload_fields(payload: dict[str, Any], upstream_payload: dict[str, Any]) -> dict[str, Any]:
    tools = normalize_upstream_tools(payload.get("tools"))
    if tools:
        upstream_payload["tools"] = tools
        upstream_payload["tool_choice"] = payload.get("tool_choice", "auto")
        if "parallel_tool_calls" in payload:
            upstream_payload["parallel_tool_calls"] = bool(payload.get("parallel_tool_calls"))
    reasoning = payload.get("reasoning")
    if isinstance(reasoning, dict):
        upstream_payload["reasoning"] = normalize_reasoning(reasoning)
        upstream_payload["include"] = payload.get("include") or ["reasoning.encrypted_content"]
    elif DEFAULT_REASONING in {"low", "medium", "high", "xhigh"}:
        upstream_payload["reasoning"] = {"effort": DEFAULT_REASONING, "summary": "auto"}
        upstream_payload["include"] = ["reasoning.encrypted_content"]
    if isinstance(payload.get("include"), list):
        upstream_payload["include"] = payload["include"]
    return upstream_payload

def iter_codex_events(upstream_payload: dict[str, Any], timeout: float = UPSTREAM_TIMEOUT) -> Generator[dict[str, Any], None, None]:
    tokens = resolve_runtime_tokens()
    try:
        yield from _iter_codex_events_with_tokens(upstream_payload, tokens, timeout)
    except urllib.error.HTTPError as exc:
        if exc.code == 401 and tokens.source == "private-store" and tokens.refresh_token:
            tokens = refresh_tokens(tokens, PRIVATE_AUTH_FILE)
            yield from _iter_codex_events_with_tokens(upstream_payload, tokens, timeout)
            return
        body = exc.read().decode("utf-8", errors="replace")
        raise OAuthAPIError(f"Codex request failed with HTTP {exc.code}: {body}", status_code=502) from exc
    except urllib.error.URLError as exc:
        raise OAuthAPIError(f"Codex request failed: {exc}", status_code=502) from exc

def _iter_codex_events_with_tokens(
    upstream_payload: dict[str, Any],
    tokens: Tokens,
    timeout: float,
) -> Generator[dict[str, Any], None, None]:
    request = urllib.request.Request(
        CODEX_RESPONSES_URL,
        data=json.dumps(upstream_payload).encode("utf-8"),
        headers=codex_headers(tokens),
        method="POST",
    )
    with urllib.request.urlopen(request, timeout=timeout) as response:
        for raw_line in response:
            line = raw_line.decode("utf-8", errors="replace").strip()
            if not line.startswith("data: "):
                continue
            if line == "data: [DONE]":
                break
            try:
                event = json.loads(line[6:])
            except json.JSONDecodeError:
                continue
            yield event

def complete_direct_responses_api(upstream_payload: dict[str, Any]) -> dict[str, Any]:
    final_response: dict[str, Any] = {}
    collected_items: list[dict[str, Any]] = []
    streamed_parts: list[str] = []
    for event in iter_codex_events(upstream_payload):
        event_type = event.get("type")
        if event_type == "response.output_text.delta":
            delta = event.get("delta")
            if isinstance(delta, str):
                streamed_parts.append(delta)
        elif event_type == "response.output_item.done":
            item = event.get("item")
            if isinstance(item, dict):
                collected_items.append(item)
        elif event_type == "response.completed":
            response_obj = event.get("response")
            if isinstance(response_obj, dict):
                final_response = response_obj
        elif event_type in {"response.failed", "response.incomplete"}:
            response_obj = event.get("response")
            raise OAuthAPIError(
                f"Codex returned {event_type}: {json.dumps(response_obj, ensure_ascii=False)}",
                status_code=502,
            )
    if not final_response:
        final_response = synthesize_direct_response(upstream_payload, "".join(streamed_parts), collected_items)
    patch_direct_response(final_response, "".join(streamed_parts), collected_items)
    return final_response

def synthesize_direct_response(upstream_payload: dict[str, Any], text: str, collected_items: list[dict[str, Any]]) -> dict[str, Any]:
    output = collected_items or [
        {
            "type": "message",
            "id": f"msg_{uuid.uuid4().hex}",
            "status": "completed",
            "role": "assistant",
            "content": [{"type": "output_text", "text": text, "annotations": []}],
        }
    ]
    return {
        "id": f"resp_{uuid.uuid4().hex}",
        "object": "response",
        "created_at": int(time.time()),
        "status": "completed",
        "model": upstream_payload.get("model", DEFAULT_MODEL_ALIAS),
        "output": output,
        "output_text": text,
    }

def patch_direct_response(response: dict[str, Any], streamed_text: str, collected_items: list[dict[str, Any]]) -> None:
    output = response.get("output")
    if (not isinstance(output, list) or not output) and collected_items:
        response["output"] = collected_items
    if not isinstance(response.get("output_text"), str) or not response.get("output_text"):
        response["output_text"] = streamed_text or extract_text_from_direct_response(response)

def extract_text_from_direct_response(response: dict[str, Any]) -> str:
    output_text = response.get("output_text")
    if isinstance(output_text, str) and output_text:
        return output_text
    output = response.get("output")
    if not isinstance(output, list):
        return ""
    parts: list[str] = []
    for item in output:
        if not isinstance(item, dict) or item.get("type") != "message":
            continue
        content = item.get("content")
        if not isinstance(content, list):
            continue
        for block in content:
            if isinstance(block, dict) and block.get("type") == "output_text" and isinstance(block.get("text"), str):
                parts.append(block["text"])
    return "".join(parts)

def extract_function_calls_from_direct_response(response: dict[str, Any]) -> list[dict[str, Any]]:
    output = response.get("output")
    if not isinstance(output, list):
        return []
    calls = []
    for item in output:
        if not isinstance(item, dict) or item.get("type") != "function_call":
            continue
        calls.append(
            {
                "id": item.get("call_id") or item.get("id") or f"call_{uuid.uuid4().hex[:24]}",
                "type": "function",
                "function": {
                    "name": item.get("name") or "",
                    "arguments": item.get("arguments") or "{}",
                },
            }
        )
    return calls

def build_direct_chat_completion_response(response: dict[str, Any], requested_model: str) -> dict[str, Any]:
    tool_calls = extract_function_calls_from_direct_response(response)
    content = extract_text_from_direct_response(response)
    message: dict[str, Any] = {"role": "assistant", "content": content}
    finish_reason = "stop" if response.get("status", "completed") == "completed" else "length"
    if tool_calls:
        message["content"] = None
        message["tool_calls"] = tool_calls
        finish_reason = "tool_calls"
    return {
        "id": f"chatcmpl-{uuid.uuid4().hex}",
        "object": "chat.completion",
        "created": int(response.get("created_at") or time.time()),
        "model": requested_model,
        "choices": [{"index": 0, "message": message, "finish_reason": finish_reason}],
        "usage": convert_direct_usage_for_chat(response.get("usage")),
    }

def convert_direct_usage_for_chat(raw_usage: Any) -> dict[str, int]:
    if not isinstance(raw_usage, dict):
        return {"prompt_tokens": 0, "completion_tokens": 0, "total_tokens": 0}
    prompt = int(raw_usage.get("input_tokens") or raw_usage.get("prompt_tokens") or 0)
    completion = int(raw_usage.get("output_tokens") or raw_usage.get("completion_tokens") or 0)
    total = int(raw_usage.get("total_tokens") or prompt + completion)
    return {"prompt_tokens": prompt, "completion_tokens": completion, "total_tokens": total}

def stream_direct_responses_api(upstream_payload: dict[str, Any]) -> Generator[bytes, None, None]:
    try:
        for event in iter_codex_events(upstream_payload):
            yield sse_bytes(event)
        yield done_bytes()
    except OAuthAPIError as exc:
        yield sse_bytes({"type": "error", "error": {"message": exc.message, "type": exc.error_type}})
        yield done_bytes()

def build_tool_prompt(
    messages: list[dict[str, Any]],
    tools: list[ToolSpec],
    tool_choice: Any,
    instructions: str | None,
) -> str:
    transcript = "\n\n".join(format_message(message) for message in messages)
    rendered_tools = "\n".join(
        f"- {tool.name}: {tool.description or 'No description'}\n  Parameters JSON Schema: {json.dumps(tool.parameters, ensure_ascii=False)}"
        for tool in tools
    )
    rule = render_tool_choice(tool_choice)
    intro = [
        "You are serving an OpenAI-compatible assistant request through Codex.",
        "Return only JSON that matches the supplied schema.",
        "Use `kind=\"assistant\"` when you should answer directly.",
        "Use `kind=\"tool_calls\"` when you want the client to execute one or more tools.",
        "Every `tool_calls[].arguments` value must be a valid JSON string.",
        f"Tool choice policy: {rule}.",
    ]
    if instructions:
        intro.append(f"Additional instructions:\n{instructions}")
    intro.append("Available tools:\n" + rendered_tools)
    intro.append("Conversation transcript:\n" + transcript)
    return "\n\n".join(intro)

def tool_schema() -> dict[str, Any]:
    return {
        "type": "object",
        "additionalProperties": False,
        "required": ["kind", "content", "tool_calls"],
        "properties": {
            "kind": {"type": "string", "enum": ["assistant", "tool_calls"]},
            "content": {"type": "string"},
            "tool_calls": {
                "type": "array",
                "items": {
                    "type": "object",
                    "additionalProperties": False,
                    "required": ["id", "name", "arguments"],
                    "properties": {
                        "id": {"type": "string"},
                        "name": {"type": "string"},
                        "arguments": {"type": "string"},
                    },
                },
            },
        },
    }

def parse_structured_output(text: str) -> tuple[str, list[ToolCall]]:
    try:
        payload = json.loads(text)
    except json.JSONDecodeError:
        start = text.find("{")
        end = text.rfind("}")
        if start >= 0 and end > start:
            payload = json.loads(text[start : end + 1])
        else:
            return text, []
    kind = payload.get("kind")
    if kind == "tool_calls":
        tool_calls: list[ToolCall] = []
        for item in payload.get("tool_calls", []):
            if not isinstance(item, dict):
                continue
            tool_calls.append(
                ToolCall(
                    call_id=str(item.get("id") or f"call_{uuid.uuid4().hex[:24]}"),
                    name=str(item.get("name") or ""),
                    arguments=str(item.get("arguments") or "{}"),
                )
            )
        return "", tool_calls
    return str(payload.get("content") or ""), []

def usage_from_event(event: dict[str, Any] | None) -> dict[str, int]:
    usage = event or {}
    input_tokens = int(usage.get("inputTokens") or 0)
    output_tokens = int(usage.get("outputTokens") or 0)
    return {
        "prompt_tokens": input_tokens,
        "completion_tokens": output_tokens,
        "total_tokens": input_tokens + output_tokens,
    }

async def complete_plain_turn(prompt: str, requested_model: str, resolved_model: str) -> AssistantResult:
    content = ""
    usage: dict[str, int] = {"prompt_tokens": 0, "completion_tokens": 0, "total_tokens": 0}
    async for event in run_codex_turn(prompt, resolved_model, output_schema=None):
        if event["type"] == "delta":
            content += event["delta"]
        elif event["type"] == "completed":
            content = event["text"] or content
            usage = event["usage"]
    return AssistantResult(content=content, tool_calls=[], usage=usage, requested_model=requested_model, resolved_model=resolved_model)

async def complete_structured_turn(
    prompt: str,
    schema: dict[str, Any],
    requested_model: str,
    resolved_model: str,
) -> AssistantResult:
    raw_text = ""
    usage: dict[str, int] = {"prompt_tokens": 0, "completion_tokens": 0, "total_tokens": 0}
    async for event in run_codex_turn(prompt, resolved_model, output_schema=schema):
        if event["type"] == "delta":
            raw_text += event["delta"]
        elif event["type"] == "completed":
            raw_text = event["text"] or raw_text
            usage = event["usage"]
    content, tool_calls = parse_structured_output(raw_text)
    return AssistantResult(content=content, tool_calls=tool_calls, usage=usage, requested_model=requested_model, resolved_model=resolved_model)

async def stream_plain_responses(
    prompt: str,
    requested_model: str,
    resolved_model: str,
) -> AsyncGenerator[bytes, None]:
    rid = response_id()
    msg_id = message_id()
    created = int(time.time())
    yield sse_bytes({"type": "response.created", "response": {"id": rid, "object": "response", "created_at": created, "model": requested_model, "status": "in_progress"}})
    yield sse_bytes(
        {
            "type": "response.output_item.added",
            "response_id": rid,
            "output_index": 0,
            "item": {"type": "message", "id": msg_id, "status": "in_progress", "role": "assistant", "content": [{"type": "output_text", "text": "", "annotations": []}]},
        }
    )
    text = ""
    usage: dict[str, int] = {"prompt_tokens": 0, "completion_tokens": 0, "total_tokens": 0}
    async for event in run_codex_turn(prompt, resolved_model, output_schema=None):
        if event["type"] == "delta" and event["delta"]:
            text += event["delta"]
            yield sse_bytes(
                {
                    "type": "response.output_text.delta",
                    "response_id": rid,
                    "item_id": msg_id,
                    "output_index": 0,
                    "content_index": 0,
                    "delta": event["delta"],
                }
            )
        elif event["type"] == "completed":
            text = event["text"] or text
            usage = event["usage"]
    yield sse_bytes(
        {
            "type": "response.output_text.done",
            "response_id": rid,
            "item_id": msg_id,
            "output_index": 0,
            "content_index": 0,
            "text": text,
        }
    )
    full_item = {"type": "message", "id": msg_id, "status": "completed", "role": "assistant", "content": [{"type": "output_text", "text": text, "annotations": []}]}
    yield sse_bytes({"type": "response.output_item.done", "response_id": rid, "output_index": 0, "item": full_item})
    yield sse_bytes(
        {
            "type": "response.completed",
            "response": {
                "id": rid,
                "object": "response",
                "created_at": created,
                "status": "completed",
                "model": requested_model,
                "output": [full_item],
                "output_text": text,
                "parallel_tool_calls": False,
                "usage": {"input_tokens": usage["prompt_tokens"], "output_tokens": usage["completion_tokens"], "total_tokens": usage["total_tokens"]},

        }
    )

            yield sse_bytes({"type": "response.output_item.added", "response_id": rid, "output_index": index, "item": {**item, "arguments": ""}})
            yield sse_bytes({"type": "response.function_call_arguments.delta", "response_id": rid, "item_id": item["id"], "output_index": index, "delta": tool.arguments})
            yield sse_bytes({"type": "response.function_call_arguments.done", "response_id": rid, "item_id": item["id"], "output_index": index, "arguments": tool.arguments})
            yield sse_bytes({"type": "response.output_item.done", "response_id": rid, "output_index": index, "item": item})
        yield sse_bytes(
            {
                "type": "response.completed",
                "response": {
                    "id": rid,
                    "object": "response",
                    "created_at": created,
                    "status": "completed",
                    "model": requested_model,
                    "output": items,
                    "output_text": "",
                    "parallel_tool_calls": False,
                    "usage": {"input_tokens": result.usage["prompt_tokens"], "output_tokens": result.usage["completion_tokens"], "total_tokens": result.usage["total_tokens"]},
                },
            }
        )
    else:
        msg_id = message_id()
        item = {"type": "message", "id": msg_id, "status": "completed", "role": "assistant", "content": [{"type": "output_text", "text": result.content, "annotations": []}]}
        yield sse_bytes({"type": "response.output_item.added", "response_id": rid, "output_index": 0, "item": {**item, "content": [{"type": "output_text", "text": "", "annotations": []}]}})
        yield sse_bytes({"type": "response.output_text.delta", "response_id": rid, "item_id": msg_id, "output_index": 0, "content_index": 0, "delta": result.content})
        yield sse_bytes({"type": "response.output_text.done", "response_id": rid, "item_id": msg_id, "output_index": 0, "content_index": 0, "text": result.content})
        yield sse_bytes({"type": "response.output_item.done", "response_id": rid, "output_index": 0, "item": item})
        yield sse_bytes(
            {
                "type": "response.completed",
                "response": {
                    "id": rid,
                    "object": "response",
                    "created_at": created,
                    "status": "completed",
                    "model": requested_model,
                    "output": [item],
                    "output_text": result.content,
                    "parallel_tool_calls": False,
                    "usage": {"input_tokens": result.usage["prompt_tokens"], "output_tokens": result.usage["completion_tokens"], "total_tokens": result.usage["total_tokens"]},
                },
            }
        )
    yield done_bytes()

def responses_input_to_messages(raw_input: Any) -> list[dict[str, Any]]:
    if raw_input is None:
        return []
    if isinstance(raw_input, str):
        return [{"role": "user", "content": raw_input}]
    if isinstance(raw_input, list):
        messages: list[dict[str, Any]] = []
        for item in raw_input:
            if isinstance(item, str):
                messages.append({"role": "user", "content": item})
                continue
            if not isinstance(item, dict):
                continue
            if isinstance(item.get("role"), str):
                messages.append({"role": item["role"], "content": item.get("content")})
                continue
            item_type = item.get("type")
            if item_type == "message":
                messages.append({"role": item.get("role", "user"), "content": item.get("content")})
            elif item_type in {"input_text", "output_text"}:
                messages.append({"role": "user", "content": item.get("text", "")})
            elif item_type in {"function_call_output", "custom_tool_call_output"}:
                messages.append(
                    {
                        "role": "tool",
                        "tool_call_id": item.get("call_id"),
                        "name": item.get("name"),
                        "content": extract_text_content(item.get("output")),
                    }
                )
        return messages
    return [{"role": "user", "content": str(raw_input)}]

def sse_headers() -> dict[str, str]:
    return {"Cache-Control": "no-cache", "Connection": "keep-alive", "X-Accel-Buffering": "no"}

def sse_bytes(payload: dict[str, Any]) -> bytes:
    return f"data: {json.dumps(payload, ensure_ascii=False)}\n\n".encode("utf-8")

def done_bytes() -> bytes:
    return b"data: [DONE]\n\n"

async def run_codex_turn(
    prompt: str,
    resolved_model: str,
    output_schema: dict[str, Any] | None,
) -> AsyncGenerator[dict[str, Any], None]:
    if not AUTH_PATH.exists():
        raise ProxyError(f"Missing Codex auth file at {AUTH_PATH}", status_code=500)
    proc = await asyncio.create_subprocess_exec(
        "codex",
        "app-server",
        "--listen",
        "stdio://",
        stdin=asyncio.subprocess.PIPE,
        stdout=asyncio.subprocess.PIPE,
        stderr=asyncio.subprocess.PIPE,
        cwd=str(ROOT),
        limit=APP_SERVER_STDIO_LIMIT,
    )
    if proc.stdin is None or proc.stdout is None or proc.stderr is None:
        raise ProxyError("Failed to start codex app-server")
    stderr_task = asyncio.create_task(proc.stderr.read())
    next_id = 1
    usage_payload: dict[str, Any] | None = None
    final_text = ""
    try:
        await send_json_rpc(proc.stdin, next_id, "initialize", {"clientInfo": {"name": "codex-openai-proxy", "version": "0.1.0"}, "capabilities": {"experimentalApi": True}})
        await wait_for_response(proc, next_id)
        next_id += 1
        await send_json_rpc(
            proc.stdin,
            next_id,
            "thread/start",
            {
                "cwd": str(ROOT),
                "approvalPolicy": DEFAULT_APPROVAL_POLICY,
                "sandbox": DEFAULT_SANDBOX,
                "model": resolved_model,
                "personality": "none",
            },
        )
        thread_response = await wait_for_response(proc, next_id)
        thread_id = thread_response["thread"]["id"]
        next_id += 1
        turn_params: dict[str, Any] = {"threadId": thread_id, "input": [{"type": "text", "text": prompt}]}
        if output_schema is not None:
            turn_params["outputSchema"] = output_schema
        await send_json_rpc(proc.stdin, next_id, "turn/start", turn_params)
        turn_response = await wait_for_response(proc, next_id)
        turn_id = turn_response["turn"]["id"]
        while True:
            message = await read_message(proc)
            if message is None:
                break
            if "id" in message and "result" in message:
                continue
            if "id" in message and "method" in message:
                await handle_server_request(proc.stdin, message)
                continue
            method = message.get("method")
            params = message.get("params") or {}
            if method == "item/agentMessage/delta":
                delta = str(params.get("delta") or "")
                if delta:
                    final_text += delta
                    yield {"type": "delta", "delta": delta}
            elif method == "item/completed":
                item = params.get("item") or {}
                if item.get("type") == "agentMessage":
                    final_text = str(item.get("text") or final_text)
            elif method == "thread/tokenUsage/updated":
                if params.get("turnId") == turn_id:
                    usage_payload = (params.get("tokenUsage") or {}).get("last") or (params.get("tokenUsage") or {}).get("total")
            elif method == "error":
                error = params.get("error") or {}
                raise ProxyError(str(error.get("message") or "Codex backend error"), status_code=502)
            elif method == "turn/completed":
                turn = params.get("turn") or {}
                if params.get("threadId") != thread_id or turn.get("id") != turn_id:
                    continue
                if turn.get("status") != "completed":
                    error = turn.get("error") or {}
                    raise ProxyError(str(error.get("message") or "Codex turn failed"), status_code=502)
                yield {"type": "completed", "text": final_text, "usage": usage_from_event(usage_payload)}
                break
    finally:
        if proc.stdin:
            proc.stdin.close()
        if proc.returncode is None:
            proc.terminate()
            try:
                await asyncio.wait_for(proc.wait(), timeout=2)
            except asyncio.TimeoutError:
                proc.kill()
                await proc.wait()
        stderr = (await stderr_task).decode("utf-8", errors="ignore").strip()
        if proc.returncode not in (0, -15) and stderr:
            raise ProxyError(f"codex app-server exited unexpectedly: {stderr}", status_code=502)

async def handle_server_request(writer: asyncio.StreamWriter, message: dict[str, Any]) -> None:
    method = message.get("method")
    request_id = message.get("id")
    if request_id is None:
        return
    if method == "account/chatgptAuthTokens/refresh":
        auth_data = json.loads(AUTH_PATH.read_text())
        tokens = auth_data.get("tokens") or {}
        result = {"accessToken": tokens.get("access_token"), "chatgptAccountId": tokens.get("account_id")}
    elif method in {"item/commandExecution/requestApproval", "item/fileChange/requestApproval", "item/permissions/requestApproval", "execCommandApproval"}:
        result = {"decision": "denied"}
    elif method == "item/tool/requestUserInput":
        result = {"confirmed": False, "responses": []}
    elif method == "item/tool/call":
        result = {"success": False, "contentItems": [{"type": "inputText", "text": "Dynamic tool calls are not handled directly by this proxy."}]}
    else:
        result = {}
    payload = {"jsonrpc": "2.0", "id": request_id, "result": result}
    writer.write((json.dumps(payload) + "\n").encode("utf-8"))
    await writer.drain()

async def send_json_rpc(
    writer: asyncio.StreamWriter,
    request_id: int,
    method: str,
    params: dict[str, Any],
) -> None:
    payload = {"jsonrpc": "2.0", "id": request_id, "method": method, "params": params}
    writer.write((json.dumps(payload) + "\n").encode("utf-8"))
    await writer.drain()

async def wait_for_response(proc: asyncio.subprocess.Process, request_id: int) -> dict[str, Any]:
    while True:
        message = await read_message(proc)
        if message is None:
            raise ProxyError("codex app-server closed before replying", status_code=502)
        if message.get("id") == request_id and "result" in message:
            return message["result"]
        if message.get("id") == request_id and "error" in message:
            raise ProxyError(str(message["error"]), status_code=502)
        if "id" in message and "method" in message and proc.stdin is not None:
            await handle_server_request(proc.stdin, message)

async def read_message(proc: asyncio.subprocess.Process) -> dict[str, Any] | None:
    if proc.stdout is None:
        return None
    while True:
        try:
            line = await proc.stdout.readline()
        except ValueError as exc:
            raise ProxyError(
                f"codex app-server emitted a JSON-RPC message larger than the {APP_SERVER_STDIO_LIMIT} byte stdio limit",
                status_code=502,
            ) from exc
        if not line:
            return None
        stripped = line.decode("utf-8", errors="ignore").strip()
        if not stripped:
            continue
        return json.loads(stripped)

def parse_args() -> argparse.Namespace:
    parser = argparse.ArgumentParser(description="OpenAI-compatible local proxy backed by Codex credentials")
    parser.add_argument("--host", default="127.0.0.1")
    parser.add_argument("--port", type=int, default=8000)
    parser.add_argument("--reload", action="store_true")
    return parser.parse_args()

def main() -> None:
    args = parse_args()
    uvicorn.run("codex_openai_server:create_app", host=args.host, port=args.port, reload=args.reload, factory=True)

if __name__ == "__main__":
    main()
